country_event = { # Great Xuaze Revolt
	id = dahe_collapse.5
	title = dahe_collapse.5.t
	desc = dahe_collapse.5.desc
	picture = COUNTRY_COLLAPSE_eventPicture
	
	major = yes
	fire_only_once = yes
	is_triggered_only = yes

	trigger = {
		#
	}

	immediate = {
		release = A43
		A43 = { grant_independence = yes }
		every_owned_province = {
			if = {
				limit = {
					is_core = A43
				}
				cede_province = A43
			}
		}
		set_global_flag = great_xuaze_revolt
	}
	
	option = { # We have to stop this!
		name = dahe_collapse.5.a
		A43 = {
			declare_war_with_cb = {
				who = ROOT
				casus_belli = cb_independence_war
			}
			hidden_effect = {
				capital_scope = {
					build_to_forcelimit = {
						infantry = 0.75
						cavalry = 0.25
					}
				}
				random_owned_province = {
					limit = {
						has_port = yes
					}
					build_to_forcelimit = {
						heavy_ship = 0.25
						light_ship = 0.4
						transport = 0.35
					}
				}
				create_general = { tradition = 100 }
				create_admiral = { tradition = 100 }
			}
			add_treasury = 300
			add_manpower = 25
			add_stability = 1
		}
		hidden_effect = {
			A66 = { country_event = { id = dahe_collapse.6 days = 60 random = 90} }
			country_event = { id = dahe_collapse.7 days = 180 random = 180 }
			every_subject = {
				add_liberty_desire = 50
			}
		}
	}
}
